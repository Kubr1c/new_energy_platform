```python
def get_model(model_type, input_size, output_size, **kwargs):
    """
    工厂函数：根据模型类型实例化对应的神经网络模型
    
    Args:
        model_type: str, 模型类型（如 'attention_lstm', 'gru', 'transformer', 'tft'）
        input_size: int, 输入特征维度
        output_size: int, 输出维度
        **kwargs: dict, 传递给模型构造函数的额外参数
    
    Returns:
        nn.Module, 实例化的模型对象
    
    Raises:
        ValueError, 如果 model_type 不在注册表中
        ImportError, 如果 torch 不可用
    """
    if model_type not in MODEL_REGISTRY:
        available = ', '.join(MODEL_REGISTRY.keys())
        raise ValueError(f"未知的模型类型: '{model_type}'。可选: {available}")

    model_cls = MODEL_REGISTRY[model_type]['class']
    
    # 特殊处理TFT模型（函数式接口）
    if model_type == 'tft':
        return model_cls(input_size, output_size)
    else:
        return model_cls(input_size, output_size=output_size, **kwargs)
```

### 4.2 Attention-LSTM模型（完全重构）

**核心文件**：[attention\_lstm.py](file:///c:/Users/86155/OneDrive/桌面/GRA/毕业设计/new_energy/backend/models/attention_lstm.py)

#### 4.2.1 三大缺陷修复

| 缺陷            | 问题描述                               | 修复方案                                       |
| ------------- | ---------------------------------- | ------------------------------------------ |
| **注意力机制失效**   | 原来使用简单线性打分，无法真正关注关键时间步             | 改用Bahdanau加性注意力，显式建模query                  |
| **信息重叠问题**    | 在LSTM1输出上计算注意力，导致与last\_hidden信息重叠 | 改用LSTM2输出计算注意力，使用LSTM2的last\_hidden作为query |
| **Dropout失效** | 单层LSTM的dropout不生效                  | 改为双层LSTM（num\_layers=2），确保层间dropout生效      |

#### 4.2.2 BahdanauAttention实现

```python
class BahdanauAttention(nn.Module):
    """
    Bahdanau（加性）注意力机制
    
    公式：score(h_t, q) = v^T · tanh(W_h·h_t + W_q·q)
    
    特点：
    - 显式建模query（当前预测需要关注的方向）
    - 使不同时间步的打分真正依赖当前预测任务
    - 提高了注意力机制的表达能力
    """
    def __init__(self, hidden_size, query_size, attn_size=None):
        super().__init__()
        if attn_size is None:
            attn_size = hidden_size // 2
        
        # 三个线性变换层
        self.W_h = nn.Linear(hidden_size, attn_size, bias=False)  # 序列隐状态变换
        self.W_q = nn.Linear(query_size, attn_size, bias=False)  # query变换
        self.v = nn.Linear(attn_size, 1, bias=False)  # 注意力得分计算

    def forward(self, encoder_outputs, query):
        """
        前向传播
        
        Args:
            encoder_outputs: (batch, seq_len, hidden_size) 编码器输出序列
            query: (batch, query_size) 查询向量，通常是最后时刻的隐状态
        
        Returns:
            context: (batch, hidden_size) 加权上下文向量
            weights: (batch, seq_len, 1) 注意力权重
        """
        # 计算注意力能量值
        energy = torch.tanh(
            self.W_h(encoder_outputs) +              # (batch, seq_len, attn_size)
            self.W_q(query).unsqueeze(1)             # (batch, 1, attn_size)
        )                                            # (batch, seq_len, attn_size)
        
        # 计算注意力得分并归一化
        scores = self.v(energy)                     # (batch, seq_len, 1)
        weights = torch.softmax(scores, dim=1)       # (batch, seq_len, 1)
        
        # 计算加权上下文向量
        context = (weights * encoder_outputs).sum(dim=1)  # (batch, hidden_size)
        
        return context, weights
```

#### 4.2.3 完整模型架构

```python
class AttentionLSTM(nn.Module):
    """
    增强版Attention-LSTM模型
    
    架构流程：
    1. LSTM1层：三层，hidden_size=128，提取初级时序特征
    2. LSTM2层：三层，hidden_size=64，进一步提炼时序特征
    3. 多头注意力：2个BahdanauAttention头，在LSTM2输出上计算
    4. 融合层：拼接注意力上下文和最后隐状态
    5. 输出层：BatchNorm → FC(256, 64) → ReLU → Dropout → FC(64, 32) → ReLU → Dropout → FC(32, output_size)
    
    增强特性：
    - LSTM层数增至3层，提高模型容量
    - 多头注意力机制，增强表达力
    - 残差连接+层归一化，稳定训练
    """
    def __init__(self, input_size, hidden_size1=128, hidden_size2=64,
                 output_size=1, dropout=0.3, num_attention_heads=2):
        
        # LSTM1：三层LSTM
        self.lstm1 = nn.LSTM(
            input_size, hidden_size1,
            num_layers=3, batch_first=True, dropout=dropout
        )
        
        # LSTM2：三层LSTM
        self.lstm2 = nn.LSTM(
            hidden_size1, hidden_size2,
            num_layers=3, batch_first=True, dropout=dropout
        )
        
        # 多头注意力
        self.num_attention_heads = num_attention_heads
        self.attention_heads = nn.ModuleList([
            BahdanauAttention(hidden_size2, hidden_size2)
            for _ in range(num_attention_heads)
        ])

        # 输出层
        self.bn = nn.BatchNorm1d(hidden_size2 * 2 * num_attention_heads)
        self.fc1 = nn.Linear(hidden_size2 * 2 * num_attention_heads, 64)
        self.fc2 = nn.Linear(64, 32)
        self.out = nn.Linear(32, output_size)
        self.drop = nn.Dropout(dropout)
        self.layer_norm = nn.LayerNorm(32)

    def forward(self, x):
        # LSTM1 → LSTM2 → 多头注意力 → 融合 → 输出
        h1, _ = self.lstm1(x)           # (batch, seq_len, 128)
        h2, _ = self.lstm2(h1)          # (batch, seq_len, 64)
        last_hidden = h2[:, -1, :]       # (batch, 64)
        
        # 多头注意力融合
        contexts = []
        for attention in self.attention_heads:
            context, _ = attention(h2, last_hidden)
            contexts.append(torch.cat([context, last_hidden], dim=1))
        combined = torch.cat(contexts, dim=1)  # (batch, 256)
        
        # 输出层
        out = self.bn(combined)
        out = F.relu(self.fc1(out))
        out = self.drop(out)
        out = self.layer_norm(out)
        out = F.relu(self.fc2(out))
        out = self.drop(out)
        out = self.out(out)
        
        return out
```

### 4.3 TFT模型（新增）

**核心文件**：[tft\_model.py](file:///c:/Users/86155/OneDrive/桌面/GRA/毕业设计/new_energy/backend/models/tft_model.py)

#### 4.3.1 模型设计思想

TFT（Temporal Fusion Transformer）是一种**专门为时间序列预测设计**的Transformer变体，融合了以下技术：

| 组件              | 功能      | 说明              |
| --------------- | ------- | --------------- |
| **门控线性单元（GLU）** | 自适应特征选择 | 决定哪些特征对预测重要     |
| **门控残差网络（GRN）** | 特征处理    | 带有门控机制的残差连接     |
| **静态协变量编码**     | 处理静态特征  | 如地理位置、设备ID等不变特征 |
| **时间自注意力**      | 捕获长期依赖  | 多头注意力机制         |
| **位置编码**        | 序列顺序信息  | 可学习的参数位置编码      |

```python
class GatedLinearUnit(nn.Module):
    """门控线性单元 - 用于自适应特征选择"""
    def __init__(self, input_dim, output_dim):
        super().__init__()
        self.linear = nn.Linear(input_dim, output_dim)
        self.gate = nn.Linear(input_dim, output_dim)
    
    def forward(self, x):
        # 门控机制：linear * sigmoid(gate)
        return self.linear(x) * torch.sigmoid(self.gate(x))


class GatedResidualNetwork(nn.Module):
    """门控残差网络 - 带门控的残差连接"""
    def __init__(self, input_dim, hidden_dim, output_dim, dropout=0.1):
        super().__init__()
        self.fc1 = nn.Linear(input_dim, hidden_dim)
        self.glu = GatedLinearUnit(hidden_dim, hidden_dim)
        self.fc2 = nn.Linear(hidden_dim, output_dim)
        self.dropout = nn.Dropout(dropout)
        self.layer_norm = nn.LayerNorm(output_dim)
        self.residual = nn.Linear(input_dim, output_dim) if input_dim != output_dim else None

    def forward(self, x):
        out = F.relu(self.fc1(x))
        out = self.glu(out)
        out = self.dropout(out)
        out = self.fc2(out)
        if self.residual is not None:
            out += self.residual(x)
        return self.layer_norm(out)


class TemporalFusionTransformer(nn.Module):
    """TFT模型主体"""
    def __init__(self, input_size, static_size=0, hidden_size=64, 
                 num_heads=4, num_layers=2, dropout=0.1, output_size=1):
        super().__init__()
        
        # 静态特征处理
        if static_size > 0:
            self.static_encoder = GatedResidualNetwork(
                static_size, hidden_size, hidden_size, dropout)
            self.static_attention = GatedResidualNetwork(
                hidden_size, hidden_size, num_heads, dropout)
        
        # 时间特征编码
        self.temporal_encoder = GatedResidualNetwork(
            input_size, hidden_size, hidden_size, dropout)
        
        # Transformer编码器
        encoder_layer = nn.TransformerEncoderLayer(
            d_model=hidden_size, nhead=num_heads,
            dim_feedforward=hidden_size * 4, dropout=dropout, batch_first=True)
        self.transformer_encoder = nn.TransformerEncoder(
            encoder_layer, num_layers=num_layers)
        
        # 输出层
        self.output_gate = GatedResidualNetwork(
            hidden_size, hidden_size, hidden_size, dropout)
        self.output_linear = nn.Linear(hidden_size, output_size)

    def forward(self, x, static=None):
        batch_size, seq_len, _ = x.shape
        
        # 时间特征编码
        temporal_emb = self.temporal_encoder(x)
        
        # 静态特征融合
        if self.static_size > 0 and static is not None:
            static_emb = self.static_encoder(static)
            static_attn = self.static_attention(static_emb)
            static_attn = static_attn.unsqueeze(1).repeat(1, seq_len, 1)
            temporal_emb = temporal_emb + static_attn
        
        # Transformer编码
        transformer_out = self.transformer_encoder(temporal_emb)
        
        # 取最后一个时间步
        last_hidden = transformer_out[:, -1, :]
        
        # 输出
        out = self.output_gate(last_hidden)
        out = self.output_linear(out)
        
        return out
```

```python
class StandardLSTM(nn.Module):
    """
    标准LSTM模型 - 基准对照
    
    架构：双层LSTM + FC输出层
    特点：无注意力机制，用于与其他模型对比

                 output_size=1, dropout=0.2):

            input_size=input_size, hidden_size=hidden_size1,
            batch_first=True, dropout=dropout)

            input_size=hidden_size1, hidden_size=hidden_size2,
            batch_first=True, dropout=dropout)
        
        self.fc = nn.Linear(hidden_size2, 16)
        self.out = nn.Linear(16, output_size)
        self.dropout = nn.Dropout(dropout)

    def forward(self, x):

        lstm_out2, _ = self.lstm2(lstm_out1)  # (batch, seq_len, 32)
        last_hidden = lstm_out2[:, -1, :]  # (batch, 32)
        
        fc_out = F.relu(self.fc(last_hidden))  # (batch, 16)
        fc_out = self.dropout(fc_out)
        output = self.out(fc_out)  # (batch, output_size)
        
        return output
```

```python
class CNNLSTM(nn.Module):
    """
    CNN-LSTM混合模型
    
    架构：
    1. 四层CNN：提取局部特征（Conv1D + BatchNorm）
    2. 双层LSTM：捕获时序依赖
    3. FC输出层：生成预测
    
    特点：
    - CNN擅长提取局部模式（功率突变、日内波动）
    - LSTM擅长捕获长期依赖
    - 跳跃连接保留低层特征
    """
    def __init__(self, input_size, hidden_size1=128, hidden_size2=64,
                 output_size=1, dropout=0.3):
        super().__init__()
        
        # CNN特征提取
        self.conv1 = nn.Conv1d(input_size, 64, kernel_size=5, padding=2)
        self.conv2 = nn.Conv1d(64, 128, kernel_size=3, padding=1)
        self.conv3 = nn.Conv1d(128, 256, kernel_size=3, padding=1)
        self.conv4 = nn.Conv1d(256, 128, kernel_size=3, padding=1)
        
        # 批归一化
        self.batch_norm1 = nn.BatchNorm1d(64)
        self.batch_norm2 = nn.BatchNorm1d(128)
        self.batch_norm3 = nn.BatchNorm1d(256)
        self.batch_norm4 = nn.BatchNorm1d(128)
        
        # LSTM
        self.lstm1 = nn.LSTM(
            input_size=128, hidden_size=hidden_size1,
            num_layers=2, batch_first=True, dropout=dropout)
        
        # 输出层
        self.fc1 = nn.Linear(hidden_size1, hidden_size2)
        self.fc2 = nn.Linear(hidden_size2, 32)
        self.out = nn.Linear(32, output_size)
        self.dropout = nn.Dropout(dropout)
        self.layer_norm = nn.LayerNorm(hidden_size2)

    def forward(self, x):
        # CNN：(batch, seq_len, input_size) → (batch, input_size, seq_len)
        x = x.permute(0, 2, 1)
        
        x1 = F.relu(self.batch_norm1(self.conv1(x)))  # (batch, 64, seq_len)
        x2 = F.relu(self.batch_norm2(self.conv2(x1)))  # (batch, 128, seq_len)
        x3 = F.relu(self.batch_norm3(self.conv3(x2)))  # (batch, 256, seq_len)
        x4 = F.relu(self.batch_norm4(self.conv4(x3)))  # (batch, 128, seq_len)
        
        # 跳跃连接
        x_cnn = x4 + x2  # (batch, 128, seq_len)
        
        # LSTM：(batch, 128, seq_len) → (batch, seq_len, 128)
        x = x_cnn.permute(0, 2, 1)
        lstm_out, _ = self.lstm1(x)  # (batch, seq_len, hidden_size1)
        
        # 输出
        last_hidden = lstm_out[:, -1, :]
        out = F.relu(self.fc1(last_hidden))
        out = self.layer_norm(out)
        out = self.dropout(out)
        out = F.relu(self.fc2(out))
        out = self.dropout(out)
        out = self.out(out)
        
        return out
```

```python
class TimeSeriesTransformer(nn.Module):
    """
    Transformer时序预测模型
    
    架构：
    1. 输入投影：特征维度 → d_model维度
    2. 位置编码：正弦-余弦位置编码
    3. Transformer编码器：多头自注意力
    4. FC输出层：生成预测
    
    特点：
    - 完全基于注意力机制，无循环结构
    - 支持并行计算
    - 擅长捕获长距离依赖
    """
    def __init__(self, input_size, hidden_size1=64, hidden_size2=32,
                 output_size=1, dropout=0.2):
        super().__init__()
        
        d_model = hidden_size1
        
        # 输入投影
        self.input_projection = nn.Linear(input_size, d_model)
        
        # 位置编码
        self.positional_encoding = PositionalEncoding(d_model, dropout=dropout)
        
        # Transformer编码器
        encoder_layer = nn.TransformerEncoderLayer(
            d_model=d_model, nhead=4, dim_feedforward=128,
            dropout=dropout, batch_first=True)
        self.transformer_encoder = nn.TransformerEncoder(
            encoder_layer, num_layers=2)
        
        # 输出层
        self.fc = nn.Linear(d_model, hidden_size2)
        self.out = nn.Linear(hidden_size2, output_size)
        self.dropout = nn.Dropout(dropout)

    def forward(self, x):
        x = self.input_projection(x)  # (batch, seq_len, d_model)
        x = self.positional_encoding(x)
        x = self.transformer_encoder(x)  # (batch, seq_len, d_model)
        
        last_output = x[:, -1, :]  # 取最后一个时间步
        out = torch.relu(self.fc(last_output))
        out = self.dropout(out)
        out = self.out(out)
        
        return out
```

```python
class EarlyStopping:
    """
    早停机制
    
    当验证损失不再改善时停止训练，防止过拟合。
    
    参数：
    - patience: 等待epoch数
    - delta: 最小改善值
    - path: 模型保存路径
    """
    def __init__(self, patience=10, delta=0, path='checkpoint.pt'):
        self.patience = patience
        self.delta = delta
        self.path = path
        self.counter = 0
        self.best_score = None
        self.early_stop = False
        self.val_loss_min = np.Inf

    def __call__(self, val_loss, model):
        """
        检查是否需要早停
        
        Returns:
            bool: 是否早停
        """
        score = -val_loss
        
        if self.best_score is None:
            self.best_score = score
            self.save_checkpoint(val_loss, model)
        elif score < self.best_score + self.delta:
            self.counter += 1
            if self.counter >= self.patience:
                self.early_stop = True
        else:
            self.best_score = score
            self.save_checkpoint(val_loss, model)
            self.counter = 0
        
        return self.early_stop

    def save_checkpoint(self, val_loss, model):
        """保存最佳模型"""
        torch.save(model.state_dict(), self.path)
        self.val_loss_min = val_loss
```

```python
class LearningRateScheduler:
    """
    学习率调度器
    
    支持多种调度策略：
    - cosine: 余弦退火
    - step: 阶梯下降
    - reduce_on_plateau: 验证指标不下降时降低
    """
    def __init__(self, optimizer, mode='cosine', initial_lr=0.001, 
                 min_lr=1e-6, patience=5):
        self.optimizer = optimizer
        self.mode = mode
        self.initial_lr = initial_lr
        self.min_lr = min_lr
        
        if mode == 'cosine':
            self.scheduler = optim.lr_scheduler.CosineAnnealingLR(
                optimizer, T_max=100, eta_min=min_lr)
        elif mode == 'step':
            self.scheduler = optim.lr_scheduler.StepLR(
                optimizer, step_size=10, gamma=0.5)
        elif mode == 'reduce_on_plateau':
            self.scheduler = optim.lr_scheduler.ReduceLROnPlateau(
                optimizer, mode='min', factor=0.5, patience=patience, min_lr=min_lr)

    def step(self, metric=None):
        """更新学习率"""
        if self.mode == 'reduce_on_plateau':
            self.scheduler.step(metric)
        else:
            self.scheduler.step()

    def get_lr(self):
        """获取当前学习率"""
        return self.optimizer.param_groups[0]['lr']
```

```python
class WeightedMSELoss(nn.Module):
    """
    加权MSE损失
    
    对不同时段的预测误差给予不同权重
    高峰时段（7-9点、18-21点）权重更高
    """
    def __init__(self, peak_hours=None, peak_weight=2.0):
        super().__init__()
        self.peak_hours = peak_hours or [7, 8, 18, 19, 20]
        self.peak_weight = peak_weight
        self.mse_loss = nn.MSELoss()

    def forward(self, pred, target, hour_indices=None):
        if hour_indices is None:
            return self.mse_loss(pred, target)
        
        weights = torch.ones_like(pred)
        for hour in self.peak_hours:
            mask = (hour_indices == hour)
            weights[mask] = self.peak_weight
        
        return (weights * (pred - target) ** 2).mean()


class HuberLoss(nn.Module):
    """
    Huber损失
    
    对异常值更鲁棒，结合MSE和MAE的优点
    """
    def __init__(self, delta=1.0):
        super().__init__()
        self.delta = delta

    def forward(self, pred, target):
        error = pred - target
        abs_error = torch.abs(error)
        quadratic = torch.min(abs_error, torch.tensor(self.delta))
        linear = abs_error - quadratic
        loss = 0.5 * quadratic ** 2 + self.delta * linear
        return loss.mean()


class CombinedLoss(nn.Module):
    """
    组合损失
    
    结合多种损失函数
    """
    def __init__(self, loss_functions, weights=None):
        super().__init__()
        self.loss_functions = loss_functions
        self.weights = weights or [1.0] * len(loss_functions)

    def forward(self, pred, target, **kwargs):
        total_loss = 0
        for loss_fn, weight in zip(self.loss_functions, self.weights):
            loss = loss_fn(pred, target)
            total_loss += weight * loss
        return total_loss
```

```python
def train_model(model, train_loader, val_loader, loss_fn, optimizer, scheduler,
               num_epochs=100, patience=10, gradient_clip=1.0, device='cpu'):
    """
    完整训练循环
    
    包含：
    - 早停机制
    - 学习率调度
    - 梯度裁剪
    - 训练/验证指标记录
    """
    early_stopping = EarlyStopping(patience=patience, 
                                  path=f'{model.__class__.__name__}_best.pt')
    
    history = {
        'train_loss': [], 'val_loss': [],
        'train_mae': [], 'val_mae': []
    }
    
    for epoch in range(num_epochs):
        # 训练阶段
        model.train()
        train_loss, train_mae = 0, 0
        
        for batch_X, batch_y, batch_hours in train_loader:
            batch_X = batch_X.to(device)
            batch_y = batch_y.to(device)
            
            optimizer.zero_grad()
            outputs = model(batch_X)
            loss = loss_fn(outputs, batch_y)
            loss.backward()
            
            # 梯度裁剪
            if gradient_clip > 0:
                torch.nn.utils.clip_grad_norm_(model.parameters(), gradient_clip)
            
            optimizer.step()
            
            train_loss += loss.item()
            train_mae += torch.mean(torch.abs(outputs - batch_y)).item()
        
        # 验证阶段
        model.eval()
        val_loss, val_mae = 0, 0
        
        with torch.no_grad():
            for batch_X, batch_y, batch_hours in val_loader:
                batch_X = batch_X.to(device)
                batch_y = batch_y.to(device)
                
                outputs = model(batch_X)
                loss = loss_fn(outputs, batch_y)
                
                val_loss += loss.item()
                val_mae += torch.mean(torch.abs(outputs - batch_y)).item()
        
        # 更新学习率
        scheduler.step(val_loss)
        
        # 记录历史
        history['train_loss'].append(train_loss / len(train_loader))
        history['val_loss'].append(val_loss / len(val_loader))
        
        # 早停检查
        if early_stopping(val_loss, model):
            print(f'Early stopping at epoch {epoch+1}')
            break
    
    # 加载最佳模型
    model.load_state_dict(torch.load(f'{model.__class__.__name__}_best.pt'))
    
    return history
```

```python
class ModelEnsemble:
    """
    模型集成类
    
    支持多种集成策略：
    - weighted_average: 加权平均
    - stacking: Stacking元学习
    - voting: 投票机制
    """
    
    def __init__(self, model_configs, strategy='weighted_average'):
        self.model_configs = model_configs  # [{'type': 'attention_lstm', 'weight': 0.35}, ...]
        self.strategy = strategy
        self.models = []
        self.weights = []
        self.meta_model = None

    def load_models(self, input_size, output_size, device='cpu'):
        """加载所有模型"""
        for config in self.model_configs:
            model = get_model(config['type'], input_size, output_size).to(device)
            self.models.append(model)
            self.weights.append(config.get('weight', 1.0))
        
        # 归一化权重
        total = sum(self.weights)
        self.weights = [w / total for w in self.weights]

    def predict_weighted_average(self, x):
        """加权平均预测"""
        predictions = []
        with torch.no_grad():
            for model in self.models:
                pred = model(x)
                predictions.append(pred)
        
        weighted_pred = torch.zeros_like(predictions[0])
        for pred, weight in zip(predictions, self.weights):
            weighted_pred += pred * weight
        
        return weighted_pred

    def predict_voting(self, x):
        """投票预测"""
        predictions = []
        with torch.no_grad():
            for model in self.models:
                predictions.append(model(x))
        
        return torch.stack(predictions).mean(dim=0)

    def predict(self, x):
        """集成预测"""
        if self.strategy == 'voting':
            return self.predict_voting(x)
        else:  # 默认加权平均
            return self.predict_weighted_average(x)

    def update_weights(self, performances):
        """根据性能动态更新权重"""
        performances = [max(0.001, p) for p in performances]
        total = sum(performances)
        self.weights = [p / total for p in performances]
```

## 7. 优化算法详解

### 7.1 粒子群优化（PSO）

**核心文件**：[pso.py](file:///c:/Users/86155/OneDrive/桌面/GRA/毕业设计/new_energy/backend/optimization/pso.py)

```python
class PSO:
    """
    粒子群优化算法
    
    用于求解储能调度优化问题
    """
    def __init__(self, n_particles=50, n_iterations=100, 
                 w=0.7, c1=1.5, c2=1.5):
        self.n_particles = n_particles
        self.n_iterations = n_iterations
        self.w = w           # 惯性权重
        self.c1 = c1         # 个体学习因子
        self.c2 = c2         # 群体学习因子

    def optimize(self, objective_func, bounds, constraints=None):
        """
        执行优化
        
        Args:
            objective_func: 目标函数
            bounds: 决策变量边界 [(min, max), ...]
            constraints: 约束条件函数
        
        Returns:
            best_position: 最优解
            best_fitness: 最优适应度
        """
        n_dims = len(bounds)
        
        # 初始化粒子群
        positions = np.random.uniform(
            [b[0] for b in bounds],
            [b[1] for b in bounds],
            (self.n_particles, n_dims)
        )
        velocities = np.random.uniform(-0.1, 0.1, (self.n_particles, n_dims))
        
        # 个体最优和全局最优
        p_best = positions.copy()
        p_best_fitness = [objective_func(p) for p in positions]
        g_best_idx = np.argmin(p_best_fitness)
        g_best = p_best[g_best_idx].copy()
        
        # 迭代优化
        for _ in range(self.n_iterations):
            for i in range(self.n_particles):
                # 更新速度
                r1, r2 = np.random.rand(n_dims), np.random.rand(n_dims)
                velocities[i] = (self.w * velocities[i] +
                                self.c1 * r1 * (p_best[i] - positions[i]) +
                                self.c2 * r2 * (g_best - positions[i]))
                
                # 更新位置
                positions[i] += velocities[i]
                
                # 边界处理
                for d in range(n_dims):
                    positions[i, d] = np.clip(positions[i, d], 
                                              bounds[d][0], bounds[d][1])
                
                # 评估适应度
                fitness = objective_func(positions[i])
                
                # 更新个体最优
                if fitness < p_best_fitness[i]:
                    p_best[i] = positions[i].copy()
                    p_best_fitness[i] = fitness
                    
                    # 更新全局最优
                    if fitness < p_best_fitness[g_best_idx]:
                        g_best_idx = i
                        g_best = positions[i].copy()
        
        return g_best, p_best_fitness[g_best_idx]
```

### 7.2 遗传算法（GA）

**核心文件**：[ga.py](file:///c:/Users/86155/OneDrive/桌面/GRA/毕业设计/new_energy/backend/optimization/ga.py)

```python
class GA:
    """
    遗传算法
    
    用于求解复杂的优化问题
    """
    def __init__(self, n_population=50, n_iterations=100,
                 crossover_rate=0.8, mutation_rate=0.1):
        self.n_population = n_population
        self.n_iterations = n_iterations
        self.crossover_rate = crossover_rate
        self.mutation_rate = mutation_rate

    def optimize(self, objective_func, bounds):
        """执行遗传算法优化"""
        n_dims = len(bounds)
        
        # 初始化种群
        population = np.random.uniform(
            [b[0] for b in bounds],
            [b[1] for b in bounds],
            (self.n_population, n_dims)
        )
        
        # 评估适应度
        fitness = [objective_func(ind) for ind in population]
        
        for _ in range(self.n_iterations):
            # 选择
            selected_idx = self._select(fitness)
            selected = population[selected_idx]
            
            # 交叉
            offspring = []
            for i in range(0, self.n_population, 2):
                if np.random.rand() < self.crossover_rate:
                    child1, child2 = self._crossover(selected[i], selected[i+1])
                else:
                    child1, child2 = selected[i], selected[i+1]
                offspring.extend([child1, child2])
            
            # 变异
            for i in range(len(offspring)):
                if np.random.rand() < self.mutation_rate:
                    offspring[i] = self._mutate(offspring[i], bounds)
            
            population = np.array(offspring[:self.n_population])
            fitness = [objective_func(ind) for ind in population]
        
        # 返回最优解
        best_idx = np.argmin(fitness)
        return population[best_idx], fitness[best_idx]

    def _select(self, fitness):
        """轮盘赌选择"""
        pass  # 简化实现

    def _crossover(self, parent1, parent2):
        """均匀交叉"""
        child1 = parent1.copy()
        child2 = parent2.copy()
        mask = np.random.rand(len(parent1)) < 0.5
        child1[mask] = parent2[mask]
        child2[mask] = parent1[mask]
        return child1, child2

    def _mutate(self, individual, bounds):
        """高斯变异"""
        for d in range(len(individual)):
            if np.random.rand() < self.mutation_rate:
                individual[d] += np.random.randn() * 0.1
                individual[d] = np.clip(individual[d], 
                                        bounds[d][0], bounds[d][1])
        return individual
```

### 7.3 自适应加权PSO（AWPSO）

**核心文件**：[awpso.py](file:///c:/Users/86155/OneDrive/桌面/GRA/毕业设计/new_energy/backend/optimization/awpso.py)

```python
class AWPSO:
    """
    自适应加权粒子群优化
    
    改进点：
    - 自适应惯性权重：根据优化进度动态调整
    - 收敛因子：加速收敛
    - 速度限制：防止粒子飞散
    """
    
    def __init__(self, n_particles=50, n_iterations=100):
        self.n_particles = n_particles
        self.n_iterations = n_iterations

    def optimize(self, objective_func, bounds):
        """执行自适应加权PSO优化"""
        n_dims = len(bounds)
        
        # 初始化（与标准PSO类似）
        positions = np.random.uniform(
            [b[0] for b in bounds],
            [b[1] for b in bounds],
            (self.n_particles, n_dims)
        )
        velocities = np.zeros((self.n_particles, n_dims))
        
        # 个体最优和全局最优
        p_best = positions.copy()
        p_best_fitness = [objective_func(p) for p in positions]
        g_best = p_best[np.argmin(p_best_fitness)].copy()
        
        # 自适应权重
        for iteration in range(self.n_iterations):
            # 计算自适应惯性权重
            w_max, w_min = 0.9, 0.4
            w = w_max - (w_max - w_min) * iteration / self.n_iterations
            
            for i in range(self.n_particles):
                # 收敛因子
                chi = 0.729 / (2 * abs(2 - 1 - 
                    np.sqrt(1 - 2 * (w - 1) / (w - 0.5))))  # 简化计算
                
                # 更新速度
                r1, r2 = np.random.rand(n_dims), np.random.rand(n_dims)
                velocities[i] = (chi * velocities[i] +
                                2.0 * r1 * (p_best[i] - positions[i]) +
                                2.0 * r2 * (g_best - positions[i]))
                
                # 速度限制
                v_max = np.array([(b[1] - b[0]) * 0.2 for b in bounds])
                velocities[i] = np.clip(velocities[i], -v_max, v_max)
                
                # 更新位置
                positions[i] += velocities[i]
                
                # 边界处理
                for d in range(n_dims):
                    positions[i, d] = np.clip(positions[i, d], 
                                              bounds[d][0], bounds[d][1])
                
                # 评估适应度
                fitness = objective_func(positions[i])
                
                if fitness < p_best_fitness[i]:
                    p_best[i] = positions[i].copy()
                    p_best_fitness[i] = fitness
                    
                    if fitness < objective_func(g_best):
                        g_best = positions[i].copy()
        
        return g_best, objective_func(g_best)
```

```sql
    start_time DATETIME NOT NULL,          -- 预测起始时间
    horizon INT NOT NULL,                  -- 预测步长：1 或 24
    data JSON,                             -- 预测数据
    mape FLOAT,                            -- 平均绝对百分比误差
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_model_type (model_type),
    INDEX idx_start_time (start_time),
    INDEX idx_created_at (created_at)
);
```
